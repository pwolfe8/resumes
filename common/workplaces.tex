%%%% Workplaces Formatting %%%%

% Define a counter for selecting the formatting style
\newcounter{workplaceformat}
% Default format
\setcounter{workplaceformat}{1}

% Command to set the workplace format
\newcommand{\setworkplaceformat}[1]{%
    \setcounter{workplaceformat}{#1}
}

% General command for formatting
\newcommand{\workplace}[3]{%
    %format1 is no dates
    %format2 is with duration (#3)
    %format3 is with dates (#2)

    \ifnum\value{workplaceformat}=1
        \noindent\textbf{#1}\par\noindent % Format 1 no dates
    \fi
    \ifnum\value{workplaceformat}=2
        \noindent\textbf{#1} \hfill \textit{#3}\par\noindent % Format 2 with duration
    \fi
    \ifnum\value{workplaceformat}=3
        \noindent\textbf{#1} \hfill \textit{#2}\par\noindent % Format 2 with dates
    \fi
}
